## 리스트(List) 다루기
- List.generate()를 활용해 숫자 리스트 생성
- removeWhere, removeLast, removeRange, clear로 리스트 요소 제거
- ... 연산자를 사용한 리스트 병합
- map()과 toList()로 타입 변환 및 필터링

  var nums = List.generate(10, (i) => i + 1);
  nums.removeWhere((n) => n % 2 == 0);

## 문자열(String) 처리
- 다양한 문자열 선언 방식: ', ", ''', """
- 문자열 보간: "${변수}" 또는 "문자열 $변수"
- toString()을 활용한 숫자 → 문자열 변환

  var result = "$x + $y = ${x * y}";

## 조건문과 불리언(Boolean)
- true, false 값 사용
- if 조건문으로 분기 처리
- 비교 연산자: <, >, ==, >=, <=
- 논리 연산자: &&, ||, !

  var isLong = author.length > maxLength;
  if (isLong) print("이름이 깁니다.");

## 반복문(Loop)
1. for 반복문 
- 일반적인 카운터 기반 반복
  for (var i = 0; i < 10; i++) {
    print(i);
  }
- 중첩 반복문 (구구단 예시)
  for (var x = 2; x <= 9; x++) {
    for (var y = 1; y <= 9; y++) {
      print("$x x $y = ${x * y}");
    }
    print('');
  }
- 리스트 순회
  var nums = [1, 2, 3];
  for (var num in nums) {
    print(num);
  }
2. while 반복문
- 조건 기반 반복
  var i = 0;
  while (i < 10) {
    print(i);
    i++;
  }
- 조건이 true일 때 한 번 실행
  var condition = true;
  while (condition) {
    print('코드실행');
    condition = false;
  }

## Set
- Set 기본 사용법
  Set<int> nums = {3, 9, 12, 41, 42, 43, 45};
  print(nums);               // 중복 없이 출력
  print(nums.length);        // 길이 확인
  print(nums.elementAt(0));  // 인덱스로 접근
- 요소 추가
  nums.add(80);                      // 하나 추가
  nums.addAll({100, 103, 105});      // 여러 개 추가
  nums.addAll([110, 111, 111, 112]); // 리스트도 추가 가능 (중복 제거됨)
- Set + 반복문
  for (var i = 0; i < nums.length; i++) {
    print(nums.elementAt(i));
  }
- Set + 조건 필터링 & 변형
  var snackNames = {'짱구','새우깡','포카칩','포스틱','감자깡','콘칩'};

  var find = snackNames.where((name) => name.length < 3).toSet();
  print(find); // 이름 길이 3 미만인 과자

  var snackNameLength = snackNames.map((name) => name.length).toSet();
  print(snackNameLength); // 이름 길이만 추출

  var customSnacks = snackNames.map((name) => "$name!").toSet();
  print(customSnacks); // 이름에 느낌표 붙이기
